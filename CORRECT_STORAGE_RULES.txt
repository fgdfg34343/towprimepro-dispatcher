rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Правила для папки drivers
    match /drivers/{driverId}/{filename} {
      // Водитель может загружать только в свою папку
      allow write: if request.auth != null && request.auth.uid == driverId;
      
      // ВАЖНО! Любой аутентифицированный может ЧИТАТЬ (для диспетчера)
      allow read: if request.auth != null;
      
      // Ограничение: файл должен быть изображением и не больше 10MB
      allow write: if request.resource.size < 10 * 1024 * 1024 &&
                      request.resource.contentType.matches('image/.*');
    }
  }
}
